`userServerID` MEDIUMINT( 0 ) DEFAULT '0' NOT NULL ,

`guestEmail` VARCHAR( 255 ) NOT NULL ,

`dateAdded` DATETIME NOT NULL ,
`dateUpdated` DATETIME NOT NULL ,
`active` TINYINT UNSIGNED DEFAULT '1' NOT NULL ,
`title` TINYINT( 255 ) NOT NULL ,
